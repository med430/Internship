% ── NestJS side ──────────────────────────────────────────────────────────────
\node[box, minimum width=5cm] (scoring) at (0, 0) {
  \textbf{ScoringService (NestJS)}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{+ computeScores(studentId)}\\
  \texttt{+ blendScores(content, semantic)}\\
  \texttt{+ buildStudentText(profile)}
};

\node[box, minimum width=5cm] (handler) at (0, -3.5) {
  \textbf{ComputeRecommendationsHandler}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{+ execute(command)}\\
  \texttt{  → for each student:}\\
  \texttt{    1. fetch candidates (content)}\\
  \texttt{    2. call IMlClient.recommend()}\\
  \texttt{    3. blend \& persist scores}
};

\node[box, minimum width=5cm] (reco_score) at (0, -7.5) {
  \textbf{RecommendationScore (Prisma)}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{studentId : UUID}\\
  \texttt{offerId : UUID}\\
  \texttt{contentScore : Float}\\
  \texttt{semanticScore : Float}\\
  \texttt{finalScore : Float}\\
  \texttt{updatedAt : DateTime}
};

\node[box, minimum width=5cm] (ml_interface) at (0, -11.5) {
  \textbf{\textlangle\textlangle interface\textrangle\textrangle\ IMlClient}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{+ recommendJobs(req) : Promise}\\
  \texttt{+ embed(text) : Promise}
};

% ── Python ML side ───────────────────────────────────────────────────────────
\node[pybox, minimum width=5cm] (fastapi) at (8, 0) {
  \textbf{FastAPI ML Service (Python)}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{POST /recommend/jobs}\\
  \texttt{POST /embed}\\
  \texttt{POST /similar}\\
  \texttt{POST /feedback}\\
  \texttt{GET  /health}
};

\node[pybox, minimum width=5cm] (embedder) at (8, -3.5) {
  \textbf{Embedder (BGE-M3)}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{model: BAAI/bge-m3}\\
  \texttt{dim: 1024}\\
  \texttt{+ encode(text) : np.ndarray}\\
  \texttt{+ cosine(u, v) : float}
};

\node[pybox, minimum width=5cm] (qdrant_svc) at (8, -7.5) {
  \textbf{QdrantService}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{collections: students, offers}\\
  \texttt{+ ensure\_collections()}\\
  \texttt{+ upsert\_vector(id, vec)}\\
  \texttt{+ retrieve\_vectors(ids)}
};

\node[pybox, minimum width=5cm] (worker) at (8, -11.5) {
  \textbf{APScheduler Worker}\\
  \rule{4.5cm}{0.4pt}\\[4pt]
  \texttt{interval: 5 min}\\
  \texttt{watermark: EmbeddingSyncState}\\
  \texttt{+ refresh\_embeddings()}\\
  \texttt{+ build\_student\_text()}\\
  \texttt{+ build\_offer\_text()}
};

  \texttt{dim: 1024}
};

% ── Formule de scoring ────────────────────────────────────────────────────────
\node[draw=gray!50, fill=gray!8, rounded corners=4pt,
